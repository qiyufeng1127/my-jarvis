## 🚀 使用指南

### 1. 设计系统
```tsx
// 在主入口引入
import '@/styles/design-system.css';

// 使用CSS变量
.my-element {
  background-color: var(--color-pink-500);
  padding: var(--spacing-4);
  border-radius: var(--radius-lg);
}
```

### 2. 基础组件
```tsx
import { Button, Card, Badge, IconBadge } from '@/components/ui';

<Button variant="primary" size="md">点击我</Button>
<Card hover shadow="lg">卡片内容</Card>
<Badge color="green" variant="soft">完成</Badge>
<IconBadge icon="💕" color="pink" size="lg" variant="soft" />
```

### 3. 手机端布局
```tsx
import { MobileBottomNav, MobileTopBar } from '@/components/layout';

// 已集成到 MobileLayout.tsx
```

### 4. 电脑端布局
```tsx
import { DesktopLayout } from '@/components/desktop';

<DesktopLayout
  sidebarItems={items}
  activeItemId={activeId}
  onItemClick={setActiveId}
  userName="用户"
  level={1}
  coins={150}
>
  <YourContent />
</DesktopLayout>
```
