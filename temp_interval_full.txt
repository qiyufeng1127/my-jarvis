                <button
                  onClick={async () => {
                    // 使用间隔开始时间作为新任务的开始时间
                    const newTask = {
                      title: '新任务',
                      scheduledStart: gap.startTime.toISOString(),
                      durationMinutes: gap.durationMinutes, // 使用间隔时长
                      taskType: 'work',
                      status: 'pending' as const,
                    };

                    }
                  }}

                    backgroundColor: isDark ? 'rgba(255,255,255,0.15)' : 'rgba(255,255,255,0.9)',

                  }}
