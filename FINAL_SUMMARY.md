## 🚀 快速开始

### 1. 引入设计系统
```tsx
// 在 main.tsx 中引入
import '@/styles/design-system.css';
```

### 2. 使用组件
```tsx
import { 
  Button, 
  Card, 
  Badge, 
  IconBadge,
  Skeleton,
  Spinner,
  EmptyState,
  ProgressBar
} from '@/components/ui';

// 使用组件
<Button variant="primary" size="md">点击我</Button>
<Card hover shadow="lg">卡片内容</Card>
<Skeleton variant="text" width="60%" />
<Spinner size="md" color="pink" />
<EmptyState icon="📝" title="还没有任务" />
<ProgressBar value={60} max={100} color="pink" />
```

### 3. 使用布局
```tsx
// 手机端
import { MobileBottomNav, MobileTopBar } from '@/components/layout';

// 电脑端
import { DesktopLayout } from '@/components/desktop';
```
